A very hackish I find it hackish too but things being hackish or impossible has never been a reason for me not to do it... way to get and insert a script and its content. I like it because what it represents in terms of physical space but I'm aware software engineers find it hackish. 

function loadScript(url, callback) {
  var script = document.createElement("script");
  script.type = "text/javascript";

  if (script.readyState) {
    script.onreadystatechange = function() {
      if (script.readyState == "loaded" || script.readyState == "complete") {
        script.onreadystatechange = null;
        callback();
      }
    };
  } else {
    script.onload = function() {
      callback();
    };
  }

  script.src = url;
  document.body.appendChild(script);
}
